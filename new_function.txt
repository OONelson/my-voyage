const selectSuggestionAndMaybePin = (suggestion: LocationSuggestion) = 
  // First select the suggestion (this will update selectedLocation and clear suggestions)  
  selectSuggestion(suggestion);  
  // Then automatically pin it if we haven't reached the limit  
  if (!reachedPinLimit.value) {  
    addPin({  
      display_name: suggestion.display_name,  
      lat: parseFloat(suggestion.lat),  
      lon: parseFloat(suggestion.lon),  
    });  
    formData.value.pins = pins.value;  
  }  
};
